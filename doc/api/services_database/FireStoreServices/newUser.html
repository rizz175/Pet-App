Future newUser({String uid, String name, DateTime dateTime, String breed}) {
  _users.doc(uid).set({
    'name': name,
    'birthday': DateFormat('yyyy-MM-dd').format(dateTime),
    'breed': breed,
    'uid': uid,
    'friends': [],
    'notification': [],
    'location': null,
  });
  return null;
}
